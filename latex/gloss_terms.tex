\newglossaryentry{MC}{
	name={metacarpal},
	description={set of bones in the palm of the human hand, shown in figure \ref{fig:hand_model:anat}}
}

\newglossaryentry{PPh}{
	name={proximal phalanx},
	plural={proximal phalanges},
	description={finger bone closest to the palm of the human hand, shown in figure \ref{fig:hand_model:anat}}
}

\newglossaryentry{DPh}{
	name={distal phalanx},
	plural={distal phalanges},
	description={finger bone farthest from the palm of the human hand, shown in figure \ref{fig:hand_model:anat}}
}


\newglossaryentry{IPh}{
	name={intermedial phalanx},
	plural={intermedial phalanges},
	description={finger bone of the human hand between \gls{PPh} and \gls{DPh}, shown in figure \ref{fig:hand_model:anat}}
}

\newglossaryentry{MPJ}{
	name={metacarpophalangeal joint},
	description={joint connecting a \gls{MC} to a \gls{PPh}}
}

\newglossaryentry{PIJ}{
	name={proximal interphalangeal joint},
	description={joint connecting a \gls{PPh} to a \gls{IPh}}
}

\newglossaryentry{DIJ}{
	name={distal interphalangeal joint},
	description={joint connecting an \gls{IPh} to a \gls{DPh}}
}

\newglossaryentry{IMU}{
	name={intertial measurement unit},
	description={technical device for measuring rotation or acceleration}
}
